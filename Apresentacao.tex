\subsection{SONET/SDH}
\frame{
\frametitle{Tecnologia SONET/SDH}
\begin{itemize}
		\item Limitações da tecnologia de transporte PDH.
		\vspace{\baselineskip}
		\item Pesquisa iniciada pela \textit{Exchange Carriers Standards Association} (ECSA):
		\begin{itemize}
				\item Início da pesquisa no ano de 1984.
				%\item Proposta para o \textit{American National Standards Institute} (ANSI).
				\item Inicio da padronização de \textit{Synchronous Optical Networking} (SONET) no ano de 1985~\cite{Ballart2002}.
		\end{itemize}
		\vspace{\baselineskip}
		\item \textit{International Telecommunication Union - Telecommunication Standardization Sector} (ITU-T).
		\begin{itemize}
				\item Interesse no padrão SONET~\cite{Boehm1990}.
				\item Propôs mudanças.
				\item Definição do padrão \textit{Synchronous Digital Hierarchy} (SDH).
		\end{itemize}
		
\end{itemize}
}

\frame{
\frametitle{Tecnologia SONET/SDH}
\begin{itemize}
		\item SONET e SDH guardam os mesmos conceitos tecnológicos.
		\vspace{\baselineskip}
		\item Principais diferenças entre SONET e SDH:
		\begin{itemize}
				\item Denominações de alguns parâmetros.
				\item Taxas de transmissão das hierarquias.
		\end{itemize}
		\vspace{\baselineskip}
		\item SONET/SDH possuem vantagens em relação ao PDH.
\end{itemize}
}

\frame{
\frametitle{Tecnologia SONET/SDH}
\begin{itemize}
		\item As vantagens da tecnologia SONET/SDH sobre PDH são:
		\begin{itemize}
				\vspace{\baselineskip}
				\item Simplificações no processo de multiplexação.
				\vspace{\baselineskip}
				\item Informações de Gerenciamento.
				\vspace{\baselineskip}
				\item Interoperabilidade.
				\vspace{\baselineskip}
				\item Disponibilidade da rede.
		\end{itemize}
\end{itemize}
}

\subsection{A rede de transporte ótica (OTN)}
\frame{
\frametitle{A rede de transporte ótica (OTN)}
\begin{itemize}
	\item Avaliação de desempenho e monitoramento da rede ótica.
	\item Estrutura de cabeçalhos simples e eficiente
	
	\item Arquitetura da OTN (Recomendação ITU-T G.872):
	
	\begin{itemize}
		\item Especifica as camadas para o monitoramento do sinal ótico.
	\end{itemize}
\end{itemize}
	
	
	\begin{figure}
		\centering
			\includegraphics[scale=.7]{cap3/otn_architecture_camadas2.png}
		\label{fig:otn_architecture_camadas}
	\end{figure}
	
	
}
\frame{
\frametitle{O padrão G.709}	

	\begin{itemize}
	\item OTN (ITU-T G.709) ou Digital Wrapper;
	\item Interfaces e taxas de dados;
	\item Transporte e monitoramento para sinais clientes;
	\item Tipos de tecnologias suportadas:
	
	\begin{itemize}
		\item SONET/SDH, ATM, Virtual Concatenator (VCAT), Generic Framing Procedure (GFP), Storage Area Network (SAN), Fibre Channel, etc.
	\end{itemize}
	
	\item Hierarquia de camadas digitais (OTH);
	\item FEC: maior alcance do sinal ótico.

		
	\end{itemize}


}

\section{Plano de Controle}
\subsection{GMPLS}
\frame{ 
\frametitle{\textit{Generalized Multiprotocol Label Switch} (GMPLS)} 
	\begin{itemize}
			\item Comutação por rótulos: Pacotes/células, Layer2 ,TDM, comprimentos de onda, interfaces óticas.
			\newline
			\item Automatizar caminhos comutados por rótulos (LSPs)
			\begin{itemize}
				\item Estabelecer, manter, remover, detectar erros e gerenciar falhas;
				\newline
				\end{itemize}
				
			\item Total separação do plano de controle do plano de dados.
			
	\end{itemize}

}


\frame{  
\frametitle{Protocolos do GMPLS} 
		\begin{itemize}
				\item Gerenciamento:
					\begin{itemize}
						\item \textit{Link Management Protocol} (LMP)
					\end{itemize}
				\vspace{\baselineskip}
				\item Roteamento:
				\begin{itemize}
					\item \textit{Open Shortest Path First Traffic Engineering} (OSPF-TE)
					\item \textit{Intermediate system to Intermediate System} (IS-IS)
				\end{itemize}
				\vspace{\baselineskip}
				\item Sinalização:
				\begin{itemize}
					\item \textit{Resource ReSerVation Protocol Traffic Engineering} (RSVP-TE)
					\item \textit{Constraint-based Label Distribution Protocol}(CR-LDP)
				\end{itemize}
		\end{itemize}
	
}

%\subsubsection{Gerenciamento}
\frame{ 
\frametitle{Protocolo de Gerenciamento: LMP} 
	\begin{itemize}
			\item Checa a presença de atividade nos planos de controle e dados entre nós adjacentes;
				\vspace{\baselineskip}
			\item Executa rotinas de recuperação para o plano de controle e o plano de dados;
				\vspace{\baselineskip}
			\item Isola locais onde ocorreram falhas no plano de dados;
				\vspace{\baselineskip}
			\item Dissemina informações sobre o estado de nós e enlaces.
			
	\end{itemize}
		
}

%\subsubsection{Roteamento}
\frame{  
\frametitle{Protocolo de Roteamento: OSPF-TE} 
	\begin{itemize}
			\item Baseado no algoritmo do menor caminho (\textit{Shortest Path First})
				\vspace{\baselineskip}
			\item Características: 
			
\begin{itemize}
	\item Considerar disponibilidade de recursos da rede (Restrições);
			\vspace{\baselineskip}
	\item Disseminar informações sobre a topologia, disponibilidade de recursos.
\end{itemize}
		\end{itemize}	
	
}

%\subsubsection{Sinalização}
\frame{ 
\frametitle{Protocolo de Sinalização: RSVP-TE} 
	
\begin{itemize}
	\item Projetado original:
	
	\begin{itemize}
		\item Reserva de recursos;
		\item Qualidade de serviço. 
	\end{itemize}	
	\vspace{\baselineskip}

	\item Foi aperfeiçoado para fazer sinalização em redes MPLS.
	\vspace{\baselineskip}
		
	\item Características da sinalização:
	\begin{itemize}
		\item Transportar parâmetros de configuração de LSPs.
		\item Alertar sobre o local e o tipo de falha ocorrida
		\item Engenharia de tráfego.
	\end{itemize}
\end{itemize}
