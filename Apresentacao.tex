\subsection{Histórico}
\frame{
\frametitle{Breve Histórico}
\begin{itemize}
		\item PDH: \textit{Plesiochronous Digital Hierarchy} (final dos anos 70 até os anos 90).
		\item SONET: \textit{Synchronous Optical Network} (meados dos anos 80 até 2000).
		\item SDH: \textit{Synchronous Digital Hierarchy} (meados dos anos 80 até 2000).
		\item ATM: \textit{Asynchronous Transfer Mode} (final dos anos 80 até 2000).
		\item Atualmente (convergência):
		\begin{itemize}
				\item OTN: \textit{Optical Transport Network}
				\item ASON: \textit{Automatically Switched Optical Network}
				\item GMPLS: \textit{Generalized Multi-Protocol Label Switching} 
		\end{itemize}
\end{itemize}
\tiny{Fonte: Slides da Disciplina de Redes Ópticas~\cite{rudge}}

\section{Protocolos de transporte}

\subsection{PDH}
\frame{
\frametitle{O protocolo PDH}
\begin{itemize}
		\item Desenvolvido para telefonia: interconexão de centrais de comutação de dados.
		\item Envio de vários sinais sobre o mesmo meio: técnicas de multiplexação no tempo com dados em formato digital.
		\item Terminais operam com relógio quase sincronizado: necessidade de inserção de bits de justificação.
\end{itemize}
}
\frame{
\frametitle{Hierarquia PDH}
\begin{itemize}
		\item Canais PDH são agrupados formando níveis hierárquicos
				\begin{itemize}
					\item 32 canais de 64 Kbps (voz) formam 1 canal com 2,048 Mbps (E1 - primeira ordem no padrão Europeu)
					\item 4 canais de primeira ordem formam 1 de segunda
				\end{itemize}
\end{itemize}
\begin{table}
\centering
\begin{tabular}{c|c|c|c}
\hline
{\bf Nível} & {\bf América do Norte} & {\bf Europa} & {\bf Japão} \\ \hline
0 & 0.064 Mb/s & 0.064 Mb/s & 0.064 Mb/s \\
1 & 1.544 Mb/s & 2.048 Mb/s & 1.544 Mb/s \\
2 & 6.321 Mb/s & 8.448 Mb/s & 6.321 Mb/s \\
3 & 44.736 Mb/s & 34.368 Mb/s & 32.064 Mb/s \\
4 & 139.264 Mb/s & 139.264 Mb/s & 97.728 Mb/s \\ \hline

\subsection{SONET/SDH}
\frame{
\frametitle{Tecnologia SONET/SDH}
\begin{itemize}
		\item Limitações da tecnologia de transporte PDH.
		\vspace{\baselineskip}
		\item Pesquisa iniciada pela \textit{Exchange Carriers Standards Association} (ECSA):
		\begin{itemize}
				\item Início da pesquisa no ano de 1984.
				%\item Proposta para o \textit{American National Standards Institute} (ANSI).
				\item Inicio da padronização de SONET no ano de 1985~\cite{Ballart2002}.
		\end{itemize}
		\vspace{\baselineskip}
		\item \textit{International Telecommunication Union - Telecommunication Standardization Sector} (ITU-T).
		\begin{itemize}
				\item Interesse no padrão SONET~\cite{Boehm1990}.
				\item Propôs mudanças.
				\item Definição do padrão SDH.
		\end{itemize}
		
\end{itemize}
}

\frame{
\frametitle{Tecnologia SONET/SDH}
\begin{itemize}
		\item SONET e SDH guardam os mesmos conceitos tecnológicos.
		\vspace{\baselineskip}
		\item Principais diferenças entre SONET e SDH:
		\begin{itemize}
				\item Denominações de alguns parâmetros.
				\item Taxas de transmissão das hierarquias.
		\end{itemize}
		\vspace{\baselineskip}
		\item SONET/SDH possuem vantagens em relação ao PDH.
\end{itemize}
}

\frame{
\frametitle{Tecnologia SONET/SDH}
\begin{itemize}
		\item As vantagens da tecnologia SONET/SDH sobre PDH são:
		\begin{itemize}
				\vspace{\baselineskip}
				\item Simplificações no processo de multiplexação.
				\vspace{\baselineskip}
				\item Informações de Gerenciamento.
				\vspace{\baselineskip}
				\item Interoperabilidade.
				\vspace{\baselineskip}
				\item Disponibilidade da rede.
		\end{itemize}
\end{itemize}
}

\subsection{OTN}

\frametitle{A Rede de Transporte Ótica (OTN)}
	\begin{itemize}
		\item Avaliação de desempenho e monitoramento da rede ótica.
		\item Estrutura de cabeçalhos simples e eficiente
	
		\item Arquitetura da OTN (Recomendação ITU-T G.872)~\cite{G872};	
		\begin{itemize}
			\item Especifica as camadas para o monitoramento do sinal ótico.
		\end{itemize}
	\end{itemize}

\frame{
\transdissolve<1>[duration=0.3]
\frametitle{Camadas da Arquitetura da OTN}
\setbeamertemplate{description item}[align left]
	\begin{description}[Canal]

		\item [Canal Ótico (OCh)]: estabelece a comunicação ótica fim a fim no domínio ótico. 
			\begin{itemize}
				\item <2> O sinal cliente é encapsulado;
				 \item<3> roteamento flexível; 
				 \item<4> cabeçalho que assegura a integridade do canal;
				 \item<5> aplica de funções de operação, administração, manutenção e provisionamento (OAM\&P), como troca de parâmetros de qualidade de serviço e funções de sobrevivência do canal ótico.
			\end{itemize}
			
	\end{description}
}

\frame{
\frametitle{Camadas da Arquitetura da OTN}
\setbeamertemplate{description item}[align left]	
	\begin{description}[Meio fisico ]	
	
		\item <1-3>[Sessão de multiplexação ótica (OMS)]: Faz mux/demux e gerencia um ou vários comprimentos de onda dentro do sinal ótico no transmissor/receptor
		\begin{itemize}
			\item <2>Processa o cabeçalho que assegura a integridade da informação de multiplexação multi-canal;
			\item <3>funções de OAM\&P da sessão de multiplexação;
		\end{itemize}
	
	\item <4-5>[Sessão de transporte ótica (OTS)]: Mídias óticas: fibras monomodo ITU-T G.652, G.653 e G.655. 
		\begin{itemize}
			\item <5>\textit{Overhead} para funções de OAM\&P;
		\end{itemize}
	\item <6> [Meio Físico]: definido pelo tipo de fibra ótica usada na rede ótica, é o servidor da transmissão ótica.
\end{description}

}

\frame{
\frametitle{Interfaces: Padrão ITU-T G.709}	

	\begin{itemize}
	\item <1-> OTN (ITU-T G.709)\cite{G709} ou \textit{Digital Wrapper};
	\item <2-> Funções de OAM\&P herdadas da experiência em Sonet/SDH.
	\item <3-> Interfaces e taxas de dados;
	\item <4-> Transporte e monitoramento para sinais clientes;
	\item <5-6>Tipos de tecnologias suportadas:
	
	\begin{itemize}
		\item <6->SONET/SDH, ATM, \textit{Virtual Concatenator} (VCAT), \textit{Generic Framing Procedure} (GFP), \textit{Storage Area Network} (SAN), \textit{Fibre Channel}, etc.
	\end{itemize}
	
	\item <7-> Hierarquia de camadas digitais (OTH);
	\item <8-> \textit{Forward Error Correction} (FEC): maior alcance do sinal ótico.
	
	\end{itemize}


}

\subsection{GMPLS}
\frame{ 
\frametitle{\textit{Generalized Multiprotocol Label Switch} (GMPLS)} 
	\begin{itemize}
			\item Comutação por rótulos: Pacotes/células, Layer2 ,TDM, comprimentos de onda, interfaces óticas.
			\newline
			\item Automatizar caminhos comutados por rótulos (LSPs)
			\begin{itemize}
				\item Estabelecer, manter, remover, detectar erros e gerenciar falhas;
				\newline
				\end{itemize}
				
			\item Total separação do plano de controle do plano de dados.
			
	\end{itemize}

}





\frame{  
\frametitle{Protocolos do GMPLS} 
		\begin{itemize}
				\item Gerenciamento:
					\begin{itemize}
						\item \textit{Link Management Protocol} (LMP)
					\end{itemize}
				\vspace{\baselineskip}
				\item Roteamento:
				\begin{itemize}
					\item \textit{Open Shortest Path First Traffic Engineering} (OSPF-TE)
					\item \textit{Intermediate system to Intermediate System} (IS-IS)
				\end{itemize}
				\vspace{\baselineskip}
				\item Sinalização:
				\begin{itemize}
					\item \textit{Resource ReSerVation Protocol Traffic Engineering} (RSVP-TE)
					\item \textit{Constraint-based Label Distribution Protocol}(CR-LDP)
				\end{itemize}
		\end{itemize}
	
}
