\section{Introdução}
\subsection{Motivação}
\frame{
\frametitle{Motivação}  

\begin{itemize} 
\item \textit{Wavelength Division Multiplexing} - WDM:
\begin{itemize}
	\item Maior capacidade de transmissão;
	\item Funções de OAM\&P: manuais, demoradas e de alto custo;
	\newline
\end{itemize}

\item Solução:
\begin{itemize}
\item \textit{Generalized Multiprotocol Label Switching} (GMPLS);
%Automatização de funções de OAM\&P;
\item \textit{Optical Transport Network - OTN};
% Melhor monitoramento e análise de desempenho;
\newline
\end{itemize}
\item Dificuldades:
\begin{itemize}
\item Poucos trabalhos sobre \textit{testbeds} com GMPLS e/ou OTN;
\item Nenhum trabalho experimental com GMPLS e OTN foi feito no Brasil;
\end{itemize}
\end{itemize} 
}
 

\subsection{Objetivos}

\frame{ 
\frametitle{Objetivos}  

\begin{itemize} 
\item Desenvolver uma aplicação de controle e monitoramento caminhos óticos:

\begin{itemize}
	\item Rede de transporte ótica ITU-T G.709;
\item Plano de controle GMPLS;
\item Considerar restrições de camada física.
\newline
\end{itemize}

\item Parâmetros:
\begin{itemize}
	\item Intensidade de potência ótica;
	\item Taxa de erros de bit (BER);
	\item Alarmes do G.709;
	\item Ganho dos amplificadores;
	\newline
\end{itemize}
\item Desafio:

\begin{itemize}
	\item Desenvolver e testar o \textit{software} em um ambiente real.
\end{itemize}

\end{itemize} } 


\section{Redes óticas}
\subsection{Conceitos Básicos}
\frame{
\frametitle{Redes óticas}

\begin{itemize}
	\item São redes nas quais os dispositivos estão conectados por meio de enlaces de fibras óticas;
	\newline
	

	\item Um caminho ótico é uma conexão fim a fim que utiliza um comprimento de onda em cada enlace ótico;
	\newline
	\item WDM:
	\newline
	\begin{itemize}
		\item Canais WDM = Comprimentos de onda;
		\item Grade do ITU-T: 80 canais ($100~GHz$);
		\newline \newline
		Exemplo: largura de banda = $3,2~Tbps$ (40~Gbps/canal)
	\end{itemize}
\end{itemize}
}
\frame{
\frametitle{Redes óticas}
Principais vantagens:
\newline
	\begin{itemize}
		\item Baixa perda ($0,2~dB/Km$ na região de $1550~nm$);
		\newline
		\item Largura de banda;
		\newline
		\item Imunidade à interferência eletromagnética externa;
		\newline
		\item Não necessita de aterramento elétrico;
		\newline
		\item Maior segurança no transporte da informação.
	\end{itemize}

}

\frame{
\frametitle{Redes Óticas: Dispositivos óticos}
\begin{itemize}
	\item Os principais dispositivos WDM são:
	
	\begin{itemize}
		\item Fibras óticas;
		\item Acopladores;
		\item Transponders;
		\item Multiplexadores e demultiplexadores (OADM e ROAMD);
		\item Comutadores óticos (OXC);
		\item Amplificadores EDFA; 
		\item Compensadores de dispersão.
		\newline
	\end{itemize}
	\item Perdas de inserção degradam o sinal ótico;

	\begin{itemize}
		\item  Diminui qualidade da transmissão.
	\end{itemize}
	
\end{itemize}
}


\frame{
\frametitle{Redes Óticas: Efeitos degenerativos}
Os efeitos degenerativos são:
\newline
\begin{itemize}
	\item Lineares: 
		\begin{itemize}
			\item Não dependem da potência do sinal;
			\item Não causam interferências entre canais WDM;
			\newline
			\newline
			Exemplos: Dispersão Cromática (CD), Dispersão Modo de Polarização (PMD), Emissão espontânea amplificada (ASE).
			\newline
		\end{itemize}
	
	\item Não-Lineares:
		\begin{itemize}
			\item Dependem da potência transmitida;
			\item Causam interferência entre canais;
			\newline
			\newline
			Exemplos: Auto modulação de fase (SPM), Modulação cruzada de fase (XPM), Mistura de Quatro ondas (FWM), Espalhamento estimulado de Brillouin (SBS) e o Espalhamento estimulado de Raman (SRS).
		\end{itemize}
	
\end{itemize}
}

\section{GMPLS}
\subsection{Conceitos Básicos}
\frame{ 
\frametitle{\textit{Generalized Multiprotocol Label Switch} (GMPLS)} 
	\begin{itemize}
			\item Comutação por rótulos: Pacotes/células, Layer2 ,TDM, comprimentos de onda, interfaces óticas.
			\newline
			\item Automatizar caminhos comutados por rótulos (LSPs)
			\begin{itemize}
				\item Estabelecer, manter, remover, detectar erros e gerenciar falhas;
				\newline
				\end{itemize}
				
			\item Total separação do plano de controle do plano de dados.
			
	\end{itemize}

}


\frame{  
\frametitle{Protocolos do GMPLS} 
		\begin{itemize}
					\item Gerenciamento:
					\begin{itemize}
						\item \textit{Link Management Protocol} (LMP)
					\end{itemize}
			\newline
			\item Roteamento:
				\begin{itemize}
					\item \textit{Open Shortest Path First Traffic Engineering} (OSPF-TE)
					\item \textit{Intermediate system to Intermediate System} (IS-IS)
				\end{itemize}
			\newline
				\item Sinalização:
				\begin{itemize}
					\item \textit{Resource ReSerVation Protocol Traffic Engineering} (RSVP-TE)
					\item \textit{Constraint-based Label Distribution Protocol}(CR-LDP)
				\end{itemize}
\end{itemize}	
}

%\subsection{Gerenciamento}
\frame{ 
\frametitle{Protocolo de Gerenciamento: LMP} 
	\begin{itemize}
			\item Checa a presença de atividade nos planos de controle e dados entre nós adjacentes;
			\newline
			\item Executa rotinas de recuperação para o plano de controle e o plano de dados;
			\newline
			\item Isola locais onde ocorreram falhas no plano de dados;
			\newline
			\item Dissemina informações sobre o estado de nós e enlaces.
			
	\end{itemize}	
}

%\subsection{Roteamento}
\frame{  
\frametitle{Protocolo de Roteamento: OSPF-TE} 
	\begin{itemize}
			\item Baseado no algoritmo do menor caminho (\textit{Shortest Path First})
			\newline
			\item Características: 
			
\begin{itemize}
	\item Considerar disponibilidade de recursos da rede (Restrições);
		\newline
	\item Disseminar informações sobre a topologia, disponibilidade de recursos.
\end{itemize}
		\end{itemize}	
	
}

\subsection{Sinalização}
\frame{ 
\frametitle{Protocolo de Sinalização: RSVP-TE} 
	
\begin{itemize}
	\item Projetado original:
	
	\begin{itemize}
		\item Reserva de recursos;
		\item Qualidade de serviço. 
	\end{itemize}
	
		\newline
	\item Foi aperfeiçoado para fazer sinalização em redes MPLS.
	\newline
	\item Características da sinalização:
	\begin{itemize}
	\item Transportar parâmetros de configuração de LSPs.
	\item Alertar sobre o local e o tipo de falha ocorrida
	\item Engenharia de tráfego.
	\end{itemize}
\end{itemize}
} 

\frame{ 
\frametitle{RSVP-TE: Estabelecimento de LSPs} 
	\begin{itemize}
			\item Mensagens \textit{Path} transporta a requisição de LSPs;
			\item Mensagem \textit{Resv} transporta o label e outros parâmetros para instalar o LSP.
	\end{itemize}		
		\newline
\begin{figure}
	\centering
		\includegraphics[scale=.3]{figuras/estabelecimento6.png}
\end{figure}
}


\frame{ 
\frametitle{RSVP-TE: Manutenção de LSPs} 
	\textit{Soft-State}: troca de mensagens frequentes.
	\begin{itemize}
			\item Mensagens \textit{Path} e \textit{Resv} de atualização.
			\item A mensagem \textit{Path} contém a rota explícita.
		\end{itemize}	
	\newline
	
\begin{figure}
	\centering
		\includegraphics[scale=.3]{figuras/manutencao2.png}
\end{figure}
} 

\frame{
 
\frametitle{RSVP-TE: Remoção de LSPs} 
	\begin{itemize}
			\item \textit{PathTear} e/ou \textit{ResvTear}: remoção de LSPs.
			\item Rota explícita.
			\item Desligamento explicito ou \textit{timeout}.
	\end{itemize}	
	\newline	
\begin{figure}
	\centering
		\includegraphics[scale=.3]{figuras/remocao3.png}

\frame{ 
\frametitle{RSVP-TE: Mensagens de Erro} 
	Erros podem acontecer devido à falhas em nós ou enlaces.
	\begin{itemize}
			\item Mensagens \textit{PathErr} e/ou \textit{ResvErr} são utilizadas.
			\item Elas notificam sobre o local e o motivo da falha.
	\end{itemize}

\frametitle{Conclusão}
\begin{itemize}
	\item Modelo de \textit{peers} óticos no plano de controle GMPLS;
	\newline
	
	\item Conhece as informações da camada ótica:
	
	\begin{itemize}
		\item intensidade de potência;
		\item ganho dos amplificadores;
	\end{itemize}
	
	\newline
	\item Capaz de monitorar os alarmes do G.709.
\end{itemize}
}

\frame{

	\newline
	\item Não altera os protocolos do GMPLS;
	\newline
	\item Três novos indicativos de erros para as mensagens \textit{NOTIFY};
		\newline
	\item Aplicação desenvolvida: controle automático, dinâmico e distribuído dos caminhos óticos.

	
	
\end{itemize}
